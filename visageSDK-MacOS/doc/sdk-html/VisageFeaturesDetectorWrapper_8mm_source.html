32 - (NSImage*)drawFeatures:(NSImage*)image
33 {
34  // if detector not initialized return the input image
35  if (visageFeaturesDetector == NULL)
36  return image;
37 
38  // convert input NSImage to IplImage
39  IplImage* input = IplImageFromNSImage(image);
40 
41  // create the output image by copying the input image
42  IplImage* output = cvCreateImage(cvGetSize(input), input->depth, input->nChannels);
43  cvCopyImage(input, output);
44 
45  // resize input image
46  float scale = (float)MAX_SIZE / MAX(input->width, input->height);
47  IplImage* resizedInput = cvCreateImage(cvSize(input->width * scale, input->height * scale), input->depth, input->nChannels);
48  cvResize(input, resizedInput);
49 
50  // detect features
51  int maxFaces = 100;
52  FaceData* data = new FaceData[maxFaces];
53  int faces = -1;
54  faces = visageFeaturesDetector->detectFacialFeatures((VsImage*)resizedInput, data, maxFaces);
55 
56  if(faces > 0)
57  {
58  // draw features into the output image
59  for(int i = 0; i < faces; i++)
60  VisageDrawing::drawResults((VsImage*)resizedInput, &data[i]);
61  }
62 
63  // convert resizedInput to 4-channels image in order to draw logo
64  IplImage* iplImageBGRA = cvCreateImage(cvSize(resizedInput->width,resizedInput->height),IPL_DEPTH_8U,4);
65  switch(resizedInput->nChannels)
66  {
67  case 4: //BGRA input image
68  cvCopy(resizedInput, iplImageBGRA);
69  break;
70  case 3: //BGR input image
71  cvCvtColor(resizedInput, iplImageBGRA, CV_BGR2BGRA);
72  break;
73  }
74 
75  // load logo image by the given path
76  NSString *logoPath = [[NSBundle mainBundle] pathForResource:@"logo" ofType:@"png"];
77  NSImage *logoImage = [[NSImage alloc] initWithContentsOfFile:logoPath];
78  IplImage* logo = IplImageFromNSImage(logoImage);
79 
80  //draw logo
81  if (logo != NULL)
82  VisageDrawing::drawLogo((VsImage*)iplImageBGRA, (VsImage*)logo);
83 
84  // convert the output IplImage to NSImage
85  NSImage* returnImage = NSImageFromIplImage(iplImageBGRA);
86 
87  // release the input and output IplImages
88  cvReleaseImage(&input);
89  cvReleaseImage(&resizedInput);
90  cvReleaseImage(&output);
91  cvReleaseImage(&iplImageBGRA);
92 
93  // return
94  return returnImage;
95 }
